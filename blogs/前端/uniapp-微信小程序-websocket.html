    const _url = "ws://localhost:9501";
import { bindSocketClient } from "../api/common.js";

// 重连对象
class ReconnectType {
  count = 0; // 重连次数
  maxCount = 10; // 重连最大次数
  interval = 5000; // 重连间隔(次/ms)
  timer = null; // 重连定时器

  // 重连
  reconnect(that) {
    clearTimeout(this.timer); // 清除定时器
    this.count++; // 重连次数+1
    // 重连
    if (this.count <= this.maxCount) {
      console.log("正在重连第" + this.count + "次");
      this.timer = setTimeout(() => that.connect(), this.interval);
    } else {
      console.log("重连次数已达上限,还是连接不上socket,放弃!");
    }
  }

  // 重置
  reset() {
    clearTimeout(this.timer); // 清除定时器
    this.count = 0; // 重连次数清零
  }
}

// 心跳对象
class HeartbeatType {
  timeout = 5000; // 心跳间隔(次/ms)
  timer = null; // 心跳定时器

  // 开启心跳
  start(that) {
    clearTimeout(this.timer); // 清除定时器
    this.timer = setTimeout(() => {
      console.log("客户端-发送心跳");
      that.send({ type: "ping", data: "ping" }); // 发送心跳
      this.start(that); // 递归
    }, this.timeout);
  }

  // 关闭心跳
  stop() {
    clearTimeout(this.timer); // 清除定时器
  }
}

// websocket对象
export class WebsocketTask {
  constructor(url = _url) {
    this.client_id = null;
    this.url = url; //连接地址
    this.socketOpen = false; //连接状态
    this.reconnect = new ReconnectType(); //重连对象
    this.heartbeat = new HeartbeatType(); //心跳对象

    // 创建连接
    this.connect();
  }

  // 创建连接
  connect() {
    console.log("创建连接 connect", this);
    uni.connectSocket({
      url: this.url,
    });
    uni.onSocketOpen((res) => {
      console.log("WebSocket 连接已打开！", res);
      this.socketOpen = true; // 连接状态
      this.reconnect.reset(); //重置重连
      this.onMessage(); // 接受服务器消息
      this.heartbeat.start(this); // 开启心跳
    });
    uni.onSocketError((res) => {
      console.log("WebSocket 连接出现错误，请检查！", res);
      this.socketOpen = false; // 连接状态
      this.reconnect.reconnect(this); // 重连
    });
    uni.onSocketClose((res) => {
      console.log("WebSocket 连接已关闭！", res);
      this.socketOpen = false; // 连接状态
      // this.reconnect.reconnect(this); // 重连
    });
  }

  // 接受服务器消息
  onMessage() {
    if (this.socketOpen) {
      uni.onSocketMessage((res) => {
        console.log("收到服务器内容：" + res.data);
        const data = JSON.parse(res.data);
        // 初始化 {"type":"init","client_id":"7f00000108fc00000019"}
        if (data.type === "init") {
          // TODO: 这里要调用接口进行用户的绑定.
          // 客户端id = data.client_id
          this.client_id = data.client_id;
          bindSocketClient({ client_id: this.client_id });
        }
        // 服务器发送给客户端的心跳 {"type":"@heart@"}
        else if (data.type === "@heart@") {
          //
        }
        // 订单结束 order_over [关闭定时器和右侧侧边的图标]
        else if (data.type === "order_over") {
          console.log("服务端消息: 订单结束");
        } else {
          // 消息错误
          console.log("接受服务器消息出现错误", data);
        }
      });
    } else {
      console.log("WebSocket连接没有打开！");
    }
  }

  // 手动发送消息
  send(data) {
    uni.sendSocketMessage({
      data: data,
      success: () => {
        console.log("发送消息给服务端成功", data);
      },
      fail: () => {
        console.log("发送消息失败", data);
        // 心跳失败
        if (data?.type === "ping") {
          this.heartbeat.stop(); //停止心跳,开启重连
          this.reconnect.reconnect(this); // 重连
        }
      },
    });
  }

  // 手动关闭连接
  close() {
    if (this.socketOpen) {
      console.log("手动关闭连接");
      uni.closeSocket();
      this.socketOpen = false; // 连接状态
      this.heartbeat.stop(); // 关闭心跳
      this.reconnect.reset(); // 重置重连
    } else {
      console.log("手动关闭连接失败，因为连接本来就已经关闭");
    }
  }
}

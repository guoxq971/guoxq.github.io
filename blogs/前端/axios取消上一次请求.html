    # 案例 - 开头直接给出代码

    可以使用 CancelToken.source 工厂方法创建 cancel token，像这样：

    const source = axios.CancelToken.source();

axios
  .get("/user/12345", {
    cancelToken: source.token,
  })
  .catch(function (thrown) {
    if (axios.isCancel(thrown)) {
      console.log("Request canceled", thrown.message);
    } else {
      // 处理错误
    }
  });

axios.post(
  "/user/12345",
  {
    name: "new name",
  },
  {
    cancelToken: source.token,
  }
);

// 取消请求（message 参数是可选的）
source.cancel("Operation canceled by the user.");

    还可以通过传递一个 executor 函数到 CancelToken 的构造函数来创建 cancel token：

    const CancelToken = axios.CancelToken;
let cancel;

axios.get("/user/12345", {
  cancelToken: new CancelToken(function executor(c) {
    // executor 函数接收一个 cancel 函数作为参数
    cancel = c;
  }),
});

// cancel the request
cancel();

    # 碰到的问题

    场景：三级联动 (图 1)

    问题：

    1. 在全选一级分类的时候，接口查询出二级和三级的数据（图 2）
      1. 比如查出三级数据1000条
    2. 这时候又查询了一次三级数据较少的一级分类（图 3）
      1. 这时候查询出来 三级数据2条
    3. 因为 1000 条 的时候可能查询了 10s，但是 2 条 的时候可能查询 1s，结果就会出现 先展示 2 条再展示 1000 条
      1. eg：图 1（默认 0 条三级数据） -> 图 2（查询出 1000 条数据，接口跑了 10s） -> 图 3（查询出 2 条数据，接口跑了 1s）
      2. 最终展示的是 1000 条数据，而不是想要的 2 调数据

    // 拦截器
const list = []; // 存储请求的数组
axios.interceptors.request.use(
  function (config) {
      //是否有取消标识
      if (config.isCancel) {

      // 生成key -start
       const key = createKek(config);
      // 生成key -end

      // 取消上一个请求 -start
      for (let item of list) {
          if (item.key === key) {
            item.cancel(); //执行取消请求
            removeCancelByKey(key); //移除数组中的请求item
        }
      }
      // 取消上一个请求 -end

      // 每个请求都有一个cancel函数 -start
      config.cancelToken = new axios.CancelToken(function(c) {
          list.push(new CancelType(key, c));
      });
      // 每个请求都有一个cancel函数 -end
    return config;
  },
  function (error) {
    return Promise.reject(error);
  }
);

axios.interceptors.response.use(
  function (response) {
    // 请求成功后，移除数组中的请求item
    removeCancelByKey(createKek(response.config));
    return response;
  },
  function (error) {
    return Promise.reject(error);
  }
);

    //上面拦截器中的方法
// 生成key
function createKek(config) {
  // 请求地址
  const requestUrl = config.url.split(".com")[1].split("?")[0];
  // 路由
  const path = vm.$route.path;
  return path + requestUrl + config.isCancel;
}
// 移除数组中的请求key
function removeCancelByKey(key) {
  for (let i = 0; i < list.length; i++) {
    if (list[i].key === key) {
      list.splice(i, 1);
      break;
    }
  }
}
/*
【独立页面】内的【接口】在【上一次请求】没回来之前【再请求】的话就【取消上一次请求】
key:路由 + 请求地址 + 标识(是否会被下一个同样的请求取消的标识 1 2 3 4)
* */
class CancelType {
  key = ""; // key
  cancel = null; // cancel函数
  constructor(key, cancel) {
    this.key = key;
    this.cancel = cancel;
  }
}
